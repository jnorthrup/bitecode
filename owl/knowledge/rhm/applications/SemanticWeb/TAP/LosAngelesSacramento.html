# When does Los Angeles play Sacramento?

# tools used: MKE, TAP, google
# Notes:
# Los Angeles and Sacramento have many sports teams.
# I picked the most likely meaning.
# hasTeamCode might have been useful for TicketMaster.


ke$ set echo=off, kbmode=tap;
ke$ ? has name = Sacramento;
  CitySacramentoCA
  MusicianSacramento_Street
  SportsTeamSacramentoKings
# has name = Sacramento
ke$ set sac = SportsTeamSacramentoKings;
ke$ $sac has ?;
# SportsTeamSacramentoKings has
  hasTeamCode = sac
  label = Sacramento Kings
  locatedIn = CitySacramentoCA
  plays = SportBasketball
  type = BasketballTeam
